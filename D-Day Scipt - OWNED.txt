-- ================== GLOW STATE ==================
local GlowStates = {} -- [UIStroke] = true/false

local function StartGlowPulse(stroke)
    if GlowStates[stroke] then return end
    GlowStates[stroke] = true

    task.spawn(function()
        while GlowStates[stroke] do
            TweenService:Create(
                stroke,
                TweenInfo.new(0.9, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut),
                { Transparency = 0.15 }
            ):Play()
            task.wait(0.9)

            if not GlowStates[stroke] then break end

            TweenService:Create(
                stroke,
                TweenInfo.new(0.9, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut),
                { Transparency = 0.55 }
            ):Play()
            task.wait(0.9)
        end
    end)
end

local function StopGlowPulse(stroke)
    GlowStates[stroke] = nil
    TweenService:Create(
        stroke,
        TweenInfo.new(0.2),
        { Transparency = 1 }
    ):Play()
end

-- ================== AIM TARGET ==================
local function getClosestEnemyInFOV()
    local closest, dist = nil, FOV_RADIUS
    local center = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)

    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Team ~= LocalPlayer.Team then
            local char = p.Character
            local part = char and char:FindFirstChild(AIM_PART_NAME)
            local hum = char and char:FindFirstChildOfClass("Humanoid")
            if part and hum and hum.Health > 0 then
                local pos, on = Camera:WorldToViewportPoint(part.Position)
                if on then
                    local d = (Vector2.new(pos.X,pos.Y)-center).Magnitude
                    if d < dist then
                        dist = d
                        closest = part
                    end
                end
            end
        end
    end
    return closest
end

local function getEquippedTool()
    local char = LocalPlayer.Character
    if not char then return nil end

    for _, v in ipairs(char:GetChildren()) do
        if v:IsA("Tool") then
            return v
        end
    end
    return nil
end

RunService.RenderStepped:Connect(function()
    local center = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
    FOVCircle.Position = center

    local aiming = UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton2)

    if not botAimEnabled then
        FOVCircle.Color = Color3.fromRGB(255,0,0)
    elseif not aiming then
        FOVCircle.Color = Color3.fromRGB(0,255,0)
    else
        FOVCircle.Color = Color3.fromRGB(120,200,255)
    end

    for _, box in pairs(ESPBoxes) do
        box.Visible = false
    end
    if not espEnabled then return end

    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Team ~= LocalPlayer.Team then
            local char = p.Character
            local hum = char and char:FindFirstChildOfClass("Humanoid")
            local hrp = char and char:FindFirstChild("HumanoidRootPart")

            if hum and hum.Health > 0 and hrp then
                local pos, on = Camera:WorldToViewportPoint(hrp.Position)
                if on then
                    createBox(p)
                    local size = math.clamp(2200/(Camera.CFrame.Position-hrp.Position).Magnitude,35,120)
                    local box = ESPBoxes[p]
                    box.Size = Vector2.new(size,size*1.6)
                    box.Position = Vector2.new(pos.X-size/2,pos.Y-size*0.8)
                    box.Visible = true
                end
            else
                removeBox(p)
            end
        end
    end
end)

        if scrollTween then
            scrollTween:Cancel()
        end

        local targetY = math.clamp(
            Content.CanvasPosition.Y + scrollVelocity,
            0,
            math.max(0, Content.CanvasSize.Y.Offset - Content.AbsoluteWindowSize.Y)
        )

        scrollTween = TweenService:Create(
            Content,
            TweenInfo.new(0.35, Enum.EasingStyle.Quint, Enum.EasingDirection.Out),

        )

        scrollTween:Play()

        scrollVelocity = 0
    end
end)

        -- 🔒 ALWAYS DEFINE FIRST
    local knobPos
    if on then
        knobPos = UDim2.fromOffset(28, 2)
    else
        knobPos = UDim2.fromOffset(2, 2)
    end

    local strokeColor = on and Color3.fromRGB(0,200,120) or Color3.fromRGB(45,45,45)

    local dir = on and Enum.EasingDirection.Out or Enum.EasingDirection.InOut

    TweenService:Create(Knob, TweenInfo.new(0.28, ease, dir), {

    }):Play()

    Button.ZIndex = 11
    Label.ZIndex = 12
    Track.ZIndex = 4
    Knob.ZIndex = 5


    -- cancel old tweens (THIS is the smoothness fix)
    if trackTween then trackTween:Cancel() end
    if knobTween then knobTween:Cancel() end
    if strokeTween then strokeTween:Cancel() end

    if animated then
        trackTween = TweenService:Create(
            Track,
            TweenInfo.new(
                0.22,                         -- faster than knob
                Enum.EasingStyle.Quart,
                Enum.EasingDirection.Out
            ),
            { BackgroundColor3 = trackColor }
        )
        trackTween:Play()


        strokeTween = TweenService:Create(
            Stroke,
            TweenInfo.new(0.25),
            { Color = strokeColor }
        )
        strokeTween:Play()

        knobTween = TweenService:Create(
            Knob,
            TweenInfo.new(
                0.32,                         -- slightly longer
                Enum.EasingStyle.Back,        -- elastic feel
                Enum.EasingDirection.Out
            ),
            {
                Position = knobPos,
                BackgroundColor3 = knobColor
            }
        )
        knobTween:Play()

    else
        Track.BackgroundColor3 = trackColor
        Stroke.Color = strokeColor
        Knob.Position = knobPos
        Knob.BackgroundColor3 = knobColor
    end

        -- glow sync (already fixed earlier)
        if on then
            startGlow()
        else
            stopGlow()
        end
    end


    Button.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then
            local newState = not stateGetter()
            stateSetter(newState)     -- flip state FIRST
            Refresh(true)             -- animate AFTER
        end
    end)


    Refresh(false)
    return Refresh
end

local AimRayParams = RaycastParams.new()
AimRayParams.FilterType = Enum.RaycastFilterType.Blacklist
AimRayParams.IgnoreWater = true

local function isTargetVisible(part)
    if not part then return false end
    local char = part.Parent
    if not char then return false end

    AimRayParams.FilterDescendantsInstances = { LocalPlayer.Character }

    local origin = Camera.CFrame.Position
    local direction = part.Position - origin
    local result = workspace:Raycast(origin, direction, AimRayParams)

    return (not result) or result.Instance:IsDescendantOf(char)
end

-- Override target selection WITH visibility check
local oldGetClosestEnemyInFOV = getClosestEnemyInFOV

function getClosestEnemyInFOV()
    local closestPart = nil
    local closestDist = FOV_RADIUS
    local screenCenter = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Team ~= LocalPlayer.Team then
            local char = player.Character
            local part = char and char:FindFirstChild(AIM_PART_NAME)
            local hum = char and char:FindFirstChildOfClass("Humanoid")

            if part and hum and hum.Health > 0 then
                local pos, onScreen = Camera:WorldToViewportPoint(part.Position)
                if onScreen then
                    local dist = (Vector2.new(pos.X, pos.Y) - screenCenter).Magnitude
                    if dist < closestDist and isTargetVisible(part) then
                        closestDist = dist
                        closestPart = part
                    end
                end
            end
        end
    end

    return closestPart
end

-- tuning
local SOFT_RELEASE = 0.25
local CONF_GAIN = 1.1
local CONF_DRAIN = 1.4
local SWITCH_RESIST = 0.55

local AimState = {
    Target = nil,
    Confidence = 0,
    LastSeen = 0
}




local lastShot = 0
local FIRE_DELAY = 0.18 -- adjust for fire rate

RunService.RenderStepped:Connect(function()
    if not autoShootEnabled then return end
    if not AimState or AimState.Target == nil then return end
    if not UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton2) then return end

    local tool = getEquippedTool()
    if not tool then return end

    local now = tick()
    if now - lastShot >= FIRE_DELAY then
        lastShot = now
        pcall(function()
            tool:Activate()
        end)
    end
end)


local RayParams = RaycastParams.new()
RayParams.FilterType = Enum.RaycastFilterType.Blacklist
RayParams.IgnoreWater = true

local function isVisible(part)
    if not part or not part.Parent then return false end
    RayParams.FilterDescendantsInstances = { LocalPlayer.Character }

    local origin = Camera.CFrame.Position
    local dir = part.Position - origin
    local hit = workspace:Raycast(origin, dir, RayParams)

    return (not hit) or hit.Instance:IsDescendantOf(part.Parent)
end

RunService:BindToRenderStep(
    "SIREN_AIM_CORE",
    Enum.RenderPriority.Camera.Value + 2,
    function(dt)
        if not botAimEnabled then
            AimState.Target = nil
            AimState.Confidence = 0
            return
        end

        if not UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton2) then
            AimState.Confidence = math.max(AimState.Confidence - CONF_DRAIN * dt, 0)
            if AimState.Confidence <= 0 then
                AimState.Target = nil
            end
            return
        end

        local candidate = getClosestEnemyInFOV()
        local now = tick()

        -- maintain lock
        if AimState.Target and isVisible(AimState.Target) then
            AimState.LastSeen = now
            AimState.Confidence = math.min(AimState.Confidence + CONF_GAIN * dt, 1)
        end

        -- acquire / switch
        if candidate and isVisible(candidate) then
            if not AimState.Target then
                AimState.Target = candidate
                AimState.LastSeen = now
            elseif candidate ~= AimState.Target and AimState.Confidence < SWITCH_RESIST then
                AimState.Target = candidate
                AimState.Confidence = 0.3
                AimState.LastSeen = now
            end
        end

        -- soft release
        if AimState.Target and not isVisible(AimState.Target) then
            if now - AimState.LastSeen > SOFT_RELEASE then
                AimState.Confidence = math.max(AimState.Confidence - CONF_DRAIN * dt, 0)
                if AimState.Confidence <= 0 then
                    AimState.Target = nil
                end
            end
        end

        -- apply camera aim
        if AimState.Target then
            local camPos = Camera.CFrame.Position
            Camera.CFrame = CFrame.new(camPos, AimState.Target.Position)
        end
    end
)
